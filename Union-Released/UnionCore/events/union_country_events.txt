# Habitable World Surveyed (HIDDEN)
ship_event = {
	id = country.52
	hide_window = yes
	
	is_triggered_only = yes
	
	trigger = {
		owner = { has_event_chain = "survey_habitable_worlds_chain" }
		FROM = {
			OR = { # Union Project
				is_planet_oceans = yes
				is_planet_deserts = yes
				is_planet_arctic = yes
				is_planet_tundra = yes
				is_planet_savannahs = yes
				is_planet_baseline = yes
				is_planet_jungle = yes
				is_planet_planetoids = yes
				is_planet_arids = yes
				is_planet_greenhousehot = yes
				is_planet_greenhousecold = yes
				is_planet_alpine = yes # Union Project
				is_planet_class = pc_gaia
			}
		}
	}
	
	immediate = {
		owner = {
			add_event_chain_counter = { 
				event_chain = "survey_habitable_worlds_chain" 
				counter = "habitable_worlds_surveyed" 
				amount = 1 
			}
		}
	}
}

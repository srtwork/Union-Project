# Start Screen Messages
# Parameters:
#   trigger, a trigger with country which decides if this is a valid part. Empty will be treated as always=yes.
#   localization, a key to a localized string.
#	location, decides where this chunk of text is located in the description, cannot be negative.

### FIRST PARTS
 
# Antarctic
part = {
	location = 0
	localization = "START_SCREEN_PART_ONE_ANTARCTIC"
	
	trigger = {
		OR = {
			ideal_planet_class = pc_CO2antarctic
			ideal_planet_class = pc_N2antarctic
			ideal_planet_class = pc_CH4antarctic
			ideal_planet_class = pc_SO2antarctic
			ideal_planet_class = pc_NH3antarctic
			ideal_planet_class = pc_CO2N2antarctic	
			ideal_planet_class = pc_antarctic		
		}
		NOT = { has_country_flag = custom_start_screen }
	}		
}

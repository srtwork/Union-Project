	}

country_event = {
	id = union_planmodif_event.1
	hide_window = yes
	is_triggered_only = yes
	fire_only_once = yes
	immediate = {
		every_planet = {
			if = {
				limit = {
					OR = { 
						is_dwarfstar = yes
						is_stdstar = yes
						is_giantstars = yes
						is_supergiantstars = yes
						is_hypergiantstars = yes
						is_binarystarsg = yes
						is_wolfstarss = yes

				set_planet_flag = union_starsone_modifier_rolled					
			}
		}			
		country_event = { id = union_planmodif_event.2 }
	}
}
